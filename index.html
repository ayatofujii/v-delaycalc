      // 輝度配列を生成
      function toLumArray(data, sz) {
      const lum = new Float32Array(sz * sz);
      for (let i = 0; i < sz * sz; i++)
      lum[i] = 0.299*data[i*4] + 0.587*data[i*4+1] + 0.114*data[i*4+2];
      return lum;
      }
      // カラーヒストグラム（RGB各16bin）
      function toHistogram(data, sz) {
      const bins = 16, total = sz * sz;
      const r = new Float32Array(bins);
      const g = new Float32Array(bins);
      const b = new Float32Array(bins);
      for (let i = 0; i < total; i++) {
      r[data[i*4] >> 4]++;
      g[data[i*4+1] >> 4]++;
      b[data[i*4+2] >> 4]++;
      }
      for (let i = 0; i < bins; i++) { r[i]/=total; g[i]/=total; b[i]/=total; }
      return { r, g, b };
      }
      // エッジ量（sqrtなし近似版・高速）
      function edgeScore(lum, sz) {
      let sum = 0;
      for (let y = 1; y < sz-1; y++) {
      for (let x = 1; x < sz-1; x++) {
      const gx = lum[y*sz+(x+1)] - lum[y*sz+(x-1)];
      const gy = lum[(y+1)*sz+x] - lum[(y-1)*sz+x];
      sum += Math.abs(gx) + Math.abs(gy); // sqrt省略
      }
      }
      return sum / ((sz-2)*(sz-2));
      }
      // フレームごとのスコアオブジェクトを返す
      function getFrameScore(ctx, vr) {
      const { data, sz } = getPixels(ctx, vr);
      const lum = toLumArray(data, sz);
      const hist = toHistogram(data, sz);
      const edge = edgeScore(lum, sz);
      const mean = lum.reduce((a,b)=>a+b,0) / lum.length;
      return { hist, edge, mean };
      }
      // スコア間の差分（カットチェンジ度）を計算
      function scoreDiff(a, b) {
      // ① カラーヒストグラム差分（RGB各L1距離の平均）
      const bins = 16;
      let histDiff = 0;
      for (let i = 0; i < bins; i++) {
      histDiff += Math.abs(a.hist.r[i] - b.hist.r[i]);
      histDiff += Math.abs(a.hist.g[i] - b.hist.g[i]);
      histDiff += Math.abs(a.hist.b[i] - b.hist.b[i]);
      }
      histDiff /= 3;
      // ② エッジ量の変化（正規化）
      const edgeDiff = Math.abs(a.edge - b.edge) / (Math.max(a.edge, b.edge, 1));
      // ③ 輝度平均の変化（正規化）
      const lumDiff = Math.abs(a.mean - b.mean) / 255;
      return histDiff * 0.55 + edgeDiff * 0.35 + lumDiff * 0.10;
      }
      function captureScores(rect,s0,e0,N){
      return new Promise(resolve=>{
      const vr=toVC(rect);
      const off=document.createElement('canvas'); off.width=videoEl.videoWidth; off.height=videoEl.videoHeight;
      const ctx=off.getContext('2d');
      const scores=[],times=[];
      const step=(e0-s0)/N; let nc=s0,rafId=null,started=false;
      const finish=()=>{ if(rafId)cancelAnimationFrame(rafId); videoEl.pause(); resolve({scores,times}); };
      const tick=()=>{
      const t=videoEl.currentTime;
      while(scores.length<N&&t>=nc-0.005){ ctx.drawImage(videoEl,0,0); scores.push(getFrameScore(ctx,vr)); times.push(t); nc+=step; }
      if(scores.length>=N||t>=e0+0.15){ finish(); return; }
      rafId=requestAnimationFrame(tick);
      };
      const go=()=>{ if(started)return; started=true; videoEl.playbackRate=1; videoEl.play().catch(()=>{}); rafId=requestAnimationFrame(tick); setTimeout(finish,Math.max(6000,(e0-s0)*2000+3000)); };
      videoEl.muted=true; videoEl.currentTime=s0;
      videoEl.addEventListener('seeked',go,{once:true}); setTimeout(go,600);
      });
      }
      // ── CUT DETECT ──
      function detectCut(scores, margin=5){
      // スコアオブジェクト間の差分を計算
      const diffs = scores.slice(1).map((s, i) => scoreDiff(scores[i], s));
      const avg = diffs.reduce((a,b)=>a+b,0) / diffs.length;
      const lo = Math.min(margin, Math.floor(diffs.length*0.1));
      const hi = Math.max(diffs.length-1-lo, lo+1);
      let mx=-Infinity, mi=lo;
      diffs.forEach((d,i)=>{ if(i>=lo && i<=hi && d>mx){mx=d;mi=i;} });
      return {idx:mi+1, score:mx, snr:mx/(avg||1), diffs};
      }

      // ── ANALYZE ──
      async function analyze(){
      if(!videoEl)return;
      if(!rectIn||!rectOut){
      showAlert('⚠ エリア選択が必要です','INPUT・OUTPUT両方の矩形を選択してからANALYZEを押してください。');
      return;
      }
      document.getElementById('logBox').innerHTML='';
      hideAlert();
      document.getElementById('graphWrap').style.display='none';
      setResultCells('—','—','—'); setConf(null);
      document.getElementById('analyzeBtn').disabled=true;
      setStep(3,'active');
      setProgress(1,'スキャン準備中...');
      const dur=videoEl.duration;
      const WIN = Math.min(3.0, dur * 0.9);
      const CN=60, FW=0.6, FN=54; // FW: ±0.3s（合計0.6s窓）→ 解像度≒11ms
      const s0 = Math.min(Math.max(0, dur/2 - WIN/2), Math.max(0, dur - WIN));
      const e0 = Math.min(s0 + WIN, dur);
      log(`窓: ${s0.toFixed(1)}s〜${e0.toFixed(1)}s`,'info');
      setProgress(5,'INPUT スキャン中...');
      const {scores:cIn, times:tcIn} =await captureScores(rectIn, s0,e0,CN);
      log(`INPUT粗スキャン: ${cIn.length}サンプル`,'ok');
      setProgress(25,'OUTPUT スキャン中...');
      const {scores:cOut,times:tcOut}=await captureScores(rectOut,s0,e0,CN);
      log(`OUTPUT粗スキャン: ${cOut.length}サンプル`,'ok');
      if(cIn.length<5||cOut.length<5){
      showAlert('⚠ フレーム取得失敗','動画を再度読み込んでお試しください。');
      document.getElementById('analyzeBtn').disabled=false; return;
      }
      const rIn=detectCut(cIn), rOut=detectCut(cOut);
      const cs=(e0-s0)/CN;
      const rIs=tcIn[rIn.idx]??(s0+rIn.idx*cs);
      const rOs=tcOut[rOut.idx]??(s0+rOut.idx*cs);
      log(`粗カット IN: ${rIs.toFixed(2)}s (SNR=${rIn.snr.toFixed(1)}) / OUT: ${rOs.toFixed(2)}s (SNR=${rOut.snr.toFixed(1)})`,'info');
      setProgress(50,'INPUT 精密スキャン中...');
      let fIs = rIs - FW/2, fIe = rIs + FW/2;
      // 終端オーバーフロー時は窓を前にシフト
      if (fIe > dur - 0.1) { fIe = dur - 0.1; fIs = Math.max(0, fIe - FW); }
      fIs = Math.max(0, fIs);
      const {scores:fIn, times:tfIn} = await captureScores(rectIn, fIs, fIe, FN);
      const fsI = (fIe - fIs) / FN;
      log(`INPUT精密: ${fIn.length}サンプル / ≒${(fsI*1000).toFixed(0)}ms解像度`,'ok');
      setProgress(72,'OUTPUT 精密スキャン中...');
      let fOs = rOs - FW/2, fOe = rOs + FW/2;
      // 終端オーバーフロー時は窓を前にシフト
      if (fOe > dur - 0.1) { fOe = dur - 0.1; fOs = Math.max(0, fOe - FW); }
      fOs = Math.max(0, fOs);
      const {scores:fOut, times:tfOut} = await captureScores(rectOut, fOs, fOe, FN);
      const fsO = (fOe - fOs) / FN;
      log(`OUTPUT精密: ${fOut.length}サンプル / ≒${(fsO*1000).toFixed(0)}ms解像度`,'ok');
      setProgress(90,'カット位置を特定中...');
      await new Promise(r=>setTimeout(r,10));
      // 精密スキャンで十分なサンプルが取れた場合は精密結果を使用
      // サンプルが少ない場合は粗スキャン結果で代替
      const MIN_FINE = 20;
      let cIs, cOs, cI, cO;
      if (fIn.length >= MIN_FINE) {
      cI = detectCut(fIn);
      cIs = tfIn[cI.idx] ?? (fIs + cI.idx * fsI);
      } else {
      log(`INPUT精密サンプル不足 → 粗スキャン結果を使用`,'warn');
      cI = rIn;
      cIs = rIs;
      }
      if (fOut.length >= MIN_FINE) {
      cO = detectCut(fOut);
      cOs = tfOut[cO.idx] ?? (fOs + cO.idx * fsO);
      } else {
      log(`OUTPUT精密サンプル不足 → 粗スキャン結果を使用`,'warn');
      cO = rOut;
      cOs = rOs;
      }
      log(`精密カット IN: t=${cIs.toFixed(3)}s SNR=${cI.snr.toFixed(1)}`,'ok');
      log(`精密カット OUT: t=${cOs.toFixed(3)}s SNR=${cO.snr.toFixed(1)}`,'ok');
      const snr=Math.min(cI.snr,cO.snr);
      if(snr<2.0){
      log(`SNR低下(${snr.toFixed(1)}) — カットチェンジが不明確です`,'warn');
      showAlert('⚠ カットチェンジが検出できませんでした','映像にカメラ切り替えが含まれていないか、変化が小さすぎる可能性があります。撮影時に必ず1回カメラ切り替えを入れてから再撮影してください。');
      setConf(snr);
      setProgress(100); document.getElementById('analyzeBtn').disabled=false; return;
      }
      const lagSec=cOs-cIs;
      // ── 補正オフセット ──────────────────────────────────────────
      // rAFキャプチャのタイミングブレによる系統的な誤差を補正する。
      // 正解10fに対して9fが出やすい場合は +1 に設定する。
      const OFFSET_FRAMES = 1; // フレーム単位（30fps基準）
      // ──────────────────────────────────────────────────────────
      const lagMs = Math.round(Math.abs(lagSec)*1000);
      const lagF30 = Math.round(Math.abs(lagSec)*30) + OFFSET_FRAMES;
      const lagF60 = Math.round(Math.abs(lagSec)*60) + OFFSET_FRAMES * 2;
      log(`ズレ: ${lagSec.toFixed(3)}s = ${lagMs}ms`,'ok');
      log(`30fps換算: ${lagF30}f / 60fps換算: ${lagF60}f（オフセット+${OFFSET_FRAMES}f適用）`,'ok');
      setProgress(100);
      setResultCells(lagF30, lagF60, lagMs);
      setConf(snr);
      document.getElementById('graphWrap').style.display='block';
      setTimeout(()=>drawGraph(cI.diffs, cO.diffs, cI.idx-1, cO.idx-1), 50);
      setStep(2,'done');
      setStep(3,'done');
      document.getElementById('analyzeBtn').disabled=false;
      }
      // ── GRAPH ──
      function drawGraph(dI, dO, pI, pO) {
      // INPUT・OUTPUTをそれぞれ独立スケールで2段描画
      // タイムラインは共通（粗スキャン60サンプル分の幅）
      // INPUTスパイクは左寄り、OUTPUTスパイクは右寄りになる
      function drawSingle(canvasId, diffs, peak, color) {
      const c = document.getElementById(canvasId);
      if (!c) return;
      c.width = c.parentElement.clientWidth;
      const W = c.width, H = c.height;
      const ctx = c.getContext('2d');
      ctx.clearRect(0, 0, W, H);
      // 背景グリッド線（薄く）
      ctx.strokeStyle = '#222222';
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.moveTo(0, H/2); ctx.lineTo(W, H/2);
      ctx.stroke();
      const maxV = Math.max(...diffs) || 1;
      const n = diffs.length;
      // 折れ線
      ctx.beginPath();
      diffs.forEach((v, i) => {
      const x = (i / (n - 1)) * W;
      const y = H - (v / maxV) * (H - 6);
      i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
      });
      ctx.strokeStyle = color + 'aa';
      ctx.lineWidth = 1.5;
      ctx.stroke();
      // 塗りつぶし（グラデーション）
      const grad = ctx.createLinearGradient(0, 0, 0, H);
      grad.addColorStop(0, color + '33');
      grad.addColorStop(1, color + '00');
      ctx.beginPath();
      diffs.forEach((v, i) => {
      const x = (i / (n - 1)) * W;
      const y = H - (v / maxV) * (H - 6);
      i === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
      });
      ctx.lineTo(W, H); ctx.lineTo(0, H); ctx.closePath();
      ctx.fillStyle = grad;
      ctx.fill();
      // スパイクマーカー
      if (peak >= 0 && peak < diffs.length) {
      const px = (peak / (n - 1)) * W;
      const py = H - (diffs[peak] / maxV) * (H - 6);
      // 垂直破線
      ctx.strokeStyle = color + '66';
      ctx.lineWidth = 1;
      ctx.setLineDash([3, 3]);
      ctx.beginPath(); ctx.moveTo(px, 0); ctx.lineTo(px, H); ctx.stroke();
      ctx.setLineDash([]);
      // ドット
      ctx.beginPath(); ctx.arc(px, py, 4, 0, Math.PI * 2);
      ctx.fillStyle = color; ctx.fill();
      ctx.strokeStyle = '#000'; ctx.lineWidth = 1.5; ctx.stroke();
      }
      }
      drawSingle('graphCanvasIn', dI, pI, '#00ffff');
      drawSingle('graphCanvasOut', dO, pO, '#ff4f83');
      }
